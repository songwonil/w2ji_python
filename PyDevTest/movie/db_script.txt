/*
테이블 생성 스크립트
*/

CREATE TABLE movie_genre
(
    `genre`     VARCHAR(45)    NOT NULL    COMMENT '장르', 
    `genre_nm`  VARCHAR(45)    NULL        COMMENT '장르명', 
    `cdate`     DATE           NULL        COMMENT '생성일', 
    PRIMARY KEY (genre)
);

ALTER TABLE movie_genre COMMENT '영화장르';

CREATE TABLE movie_item
(
    `movieid`           INT            NOT NULL    AUTO_INCREMENT COMMENT '영화id', 
    `movietitle`        VARCHAR(45)    NULL        COMMENT '제목', 
    `releasedate`       VARCHAR(8)     NULL        COMMENT '영화개봉날짜', 
    `videoreleasedate`  VARCHAR(8)     NULL        COMMENT '비디오개봉날짜', 
    `IMDBURL`           VARCHAR(45)    NULL        COMMENT '영화url', 
    `genre`             VARCHAR(45)    NULL        COMMENT '장르', 
    `cdate`             DATE           NULL        COMMENT '생성일', 
    PRIMARY KEY (movieid)
);

ALTER TABLE movie_item COMMENT '영화 테이블';

ALTER TABLE movie_item
    ADD CONSTRAINT FK_movie_item_genre_movie_genre_genre FOREIGN KEY (genre)
        REFERENCES movie_genre (genre) ON DELETE RESTRICT ON UPDATE RESTRICT;


CREATE TABLE movie_users
(
    `user_id`     VARCHAR(45)    NOT NULL    COMMENT '사용자 id', 
    `age`         VARCHAR(45)    NULL        COMMENT '나이', 
    `gender`      VARCHAR(45)    NULL        COMMENT '성별', 
    `occupation`  VARCHAR(45)    NULL        COMMENT '직업', 
    `zipcode`     VARCHAR(45)    NULL        COMMENT '우편번호', 
    `cdate`       date           NULL        COMMENT '생성일', 
    PRIMARY KEY (user_id)
);

ALTER TABLE movie_users COMMENT '영화관객';


CREATE TABLE movie_rate
(
    `rate_seq`    INT            NOT NULL    AUTO_INCREMENT COMMENT '평점seq', 
    `movieid`     VARCHAR(45)    NULL        COMMENT '영화id', 
    `user_id`     VARCHAR(45)    NULL        COMMENT '관객id', 
    `rating`      INT            NULL        COMMENT '평점', 
    `timestampp`  DATETIME       NULL        COMMENT '평점시간', 
    `cdate`       DATE           NULL        COMMENT '생성시간', 
    PRIMARY KEY (rate_seq)
);

ALTER TABLE movie_rate COMMENT '영화 평점';

ALTER TABLE movie_rate
    ADD CONSTRAINT FK_movie_rate_movieid_movie_item_movietitle FOREIGN KEY (movieid)
        REFERENCES movie_item (movietitle) ON DELETE RESTRICT ON UPDATE RESTRICT;

ALTER TABLE movie_rate
    ADD CONSTRAINT FK_movie_rate_user_id_movie_users_user_id FOREIGN KEY (user_id)
        REFERENCES movie_users (user_id) ON DELETE RESTRICT ON UPDATE RESTRICT;
